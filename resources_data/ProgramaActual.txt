void contarPulsoDer() {

// --- FUNCIÓN PARA DETENER LOS MOTORES ---
void detener() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
}

// --- FUNCIÓN PARA RECORRER UNA DISTANCIA ESPECÍFICA ---
void avanzarDistancia(float distanciaObjetivo, int velocidad) {
  // Reiniciar contadores antes de arrancar cada ciclo
  pulsosIzq = 0;
  pulsosDer = 0;

  // Calcular cuántos pulsos requerimos
  long pulsosObjetivo = distanciaObjetivo / Cm;

  Serial.print("Pulsos objetivo para este ciclo: ");
  Serial.println(pulsosObjetivo);

  // Motores hacia adelante
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
  
  // Aplicar velocidad PWM (rango 0-255)
  analogWrite(ENA, velocidad);
  analogWrite(ENB, velocidad);

  // Bucle de control: se queda aquí hasta alcanzar los pulsos meta
  while (pulsosIzq < pulsosObjetivo || pulsosDer < pulsosObjetivo) {
    Serial.print("Progreso -> Izq: "); Serial.print(pulsosIzq);
    Serial.print(" | Der: "); Serial.println(pulsosDer);
    delay(50); 
  }

  // Al salir del ciclo, frenar
  detener();
}
